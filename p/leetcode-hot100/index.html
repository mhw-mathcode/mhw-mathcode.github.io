[排序链表]

题意：对于一个链表，在 $O(nlogn)$ 时间复杂度和常数级空间复杂度下返回排序后的链表。

做法：归并排序的思路，每次先要用快慢指针找到待排链表的中间结点，然后对于每次合并，用一个虚节点作为起始结点，依次向后面连接。

class Solution {
public:
    ListNode* middle(ListNode* head) {
        ListNode* lst = head;
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast && fast->next) {
            lst = slow;
            slow = slow->next;
            fast = fast->next->next;
        }
        lst->next = nullptr;
        return slow;
    }
    ListNode* merge(ListNode* a, ListNode* b) {
        ListNode* ans = new ListNode(0, nullptr);
        ListNode* ans2 = ans;
        while (a && b) {
            if (a->val < b->val) {
                ans->next = a;
                a = a->next;
            }
            else {
                ans->next = b;
                b = b->next;
            }
            ans = ans->next;
        }
        ans->next = a ? a : b;
        return ans2->next;
    }
    ListNode* sortList(ListNode* head) {
        if (!head || !head->next) {
            return head;
        }
        ListNode* head2 = middle(head);
        head = sortList(head);
        head2 = sortList(head2);
        
        return merge(head, head2);
    }
};

class Solution {
public:
    vector<vector<int>> permute(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<vector<int>> ans;
        vector<int> t;
        do {
            t.clear();
            for (auto i: nums) t.push_back(i);
            ans.push_back(t);
        } while (next_permutation(nums.begin(), nums.end()));
        return ans;
    }
};

[搜索旋转排序数组]

题意：对于一个升序数组，将后面若干元素移到前面，在 $O(logn)$ 时间复杂度下找到一个目标值。

做法：有类似题的经验，我们已经知道该数组可以根据最后一个元素划分为两个单增数组，只需要判断 $nums[mid]$ 与目标值 $target$ 的位置关系，因此：

  • 如果 $x$ 和 $target$ 在不同的递增段：
    • 如果 $target$ 在第一段（左），$x$ 在第二段（右），说明 $x$ 在 $target$ 右边；
    • 如果 $target$ 在第二段（右），$x$ 在第一段（左），说明 $x$ 在 $target$ 左边。
  • 如果 $x$ 和 $target$ 在相同的递增段：比较 $x$ 和 $target$ 的大小即可。

class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int n = preorder.size();
        unordered_map<int, int> mp;
        // 每个点在中序遍历中的位置
        for (int i = 0; i < n; i++) mp[inorder[i]] = i;

        auto dfs = [&](auto self, int pl, int pr, int il, int ir) ->  TreeNode* {
            if (pl > pr) return nullptr;
            int lsize = mp[preorder[pl]] - il;
            TreeNode *left = self(self, pl + 1, pl + lsize, il, il + lsize - 1);
            TreeNode *right = self(self, pl + lsize + 1, pr, il + lsize + 1, ir);
            return new TreeNode(preorder[pl], left, right);
        };

        return dfs(dfs, 0, n - 1, 0, n - 1);
    }   
};

[寻找两个正序数组的中位数]

题意：对于两个有序数组，使用 $O(log(m+n))$ 的时间复杂度找到中位数。

做法：双指针 $O(n+m)$

将两个数组分为两组，保证第一组的最大值小于第二组的最小值，即可得到中位数。如果 $nums1$ 中有 $i$ 个数在第一组，那么 $nums2$ 中就有 $j=(sum+1)/2-i$ 个数在第一组，因此第一组最大值为 $max(nums1[i], nums2[j])$。剩余元素为第二组，最小值为 $min(nums1[i + 1], nums2[j + 1])$ 。因为数组本身有序，所以只需要求 $ nums1[i] < nums2[j + 1]$ 并且 $nums2[j] < nums1[i + 1]$。一个细节是，我们在 $i–,j++$ 双指针移动的过程中，如果遇到了 $nums1[i + 1] > nums2[j]$ 即结束循环，那其实一定有 $nums1[i] < nums2[j + 1]$。

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        if (nums1.size() > nums2.size()) swap(nums1, nums2);

        int sum = nums1.size() + nums2.size();
        nums1.insert(nums1.begin(), INT_MIN);
        nums2.insert(nums2.begin(), INT_MIN);
        nums1.push_back(INT_MAX);
        nums2.push_back(INT_MAX);        

        // nums1 中前 i 个数与 nums2 中前 j 个数组成第一组，最大值为 max(nums1[i], nums2[j])
        // 剩余元素为第二组，最小值为 min(nums1[i + 1], nums2[j + 1])
        int i = 0, j = (sum + 1) / 2;

        // 所以要求 nums1[i] < nums2[j + 1] 并且 nums2[j] < nums1[i + 1]
        // nums1[i + 1] > nums2[j] 的同时 nums1[i] < nums2[j + 1]
        while (nums1[i + 1] < nums2[j]) {
            i++, j--;
        }
        
        if (sum & 1) return max(nums1[i], nums2[j]);
        else return (max(nums1[i], nums2[j]) + min(nums1[i + 1], nums2[j + 1])) / 2.0;
    }
};

二分优化 $O(log(min(m, n)))$，如果可以理解上面的思路，那么其实就是在第一组中找到最小的 $i$ 使得 $nums1[i + 1] >= nums2[j]$ 。

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        if (nums1.size() > nums2.size()) swap(nums1, nums2);

        int sum = nums1.size() + nums2.size();
        nums1.insert(nums1.begin(), INT_MIN);
        nums2.insert(nums2.begin(), INT_MIN);
        nums1.push_back(INT_MAX);
        nums2.push_back(INT_MAX);        

        int l = 0, r = nums1.size() - 1, ans = r;
        while (l <= r) {
            int i = (l + r) / 2;
            int j = (sum + 1) / 2 - i; 
            if (nums1[i + 1] >= nums2[j]) r = i - 1, ans = min(ans, i);
            else l = i + 1;
        }
        int i = ans, j = (sum + 1) / 2 - i;
        
        if (sum & 1) return max(nums1[i], nums2[j]);
        else return (max(nums1[i], nums2[j]) + min(nums1[i + 1], nums2[j + 1])) / 2.0;
    }
};

[缺失的第一个正数]

题意：使用 $O(n)$ 的时间复杂度与常数级的空间复杂度找到一个数组的 $mex$。

做法：我们只需要考虑值为 $1-n$ 的数，对于其他值直接将其设置为 $n+1$。随后我们用这个数组来充当哈希表，用 $nums[i]$ 的正负表示 $i$ 有没有出现过，也就是说数组中所有值的绝对值表示真正的数值，同时对于绝对值大于 $n+1$ 的值也无需处理。最后数组中第一个为正数的位置即为答案。

class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            if (nums[i] < 1 || nums[i] > n) 
                nums[i] = n + 1;
        }
        for (int i = 0; i < n; i++) {
            if (abs(nums[i]) <= n) 
                nums[abs(nums[i]) - 1] = -abs(nums[abs(nums[i]) - 1]);
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] >= 0) 
                return i + 1;
        }
        return n + 1;
    }
};

class MedianFinder {
public:
    priority_queue<int, vector<int>, greater<int>> qmax; // 小根堆，较大的一半
    priority_queue<int, vector<int>, less<int>> qmin; // 大根堆，较小的一半
    MedianFinder() { }
    
    void addNum(int num) {
        if (qmin.size() == 0) {
            qmin.push(num);
        }
        else if (num <= qmin.top())
        {
            qmin.push(num);
            if (qmin.size() - qmax.size() >= 2) {
                int t = qmin.top();
                qmin.pop();
                qmax.push(t);
            }
        }
        else {
            qmax.push(num);
            if (qmax.size() > qmin.size()) {
                int t = qmax.top();
                qmax.pop();
                qmin.push(t);
            }
        }
    }
    
    double findMedian() {
        if (qmin.size() == qmax.size()) return (qmin.top() + qmax.top()) / 2.0;
        else return qmin.top();
    }
};

class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<int> col(n, 0), v1(2 * n, 0), v2(2 * n, 0);
        vector<string> a(n, string(n, '.'));
        vector<vector<string>> ans;
        auto dfs = [&](auto& self, int i) -> void {
            if (i == n) {
                ans.push_back(a);
                return;
            }
            for (int j = 0; j < n; j++) {
                if (col[j] == 0 && v1[i + j] == 0 && v2[i - j + n] == 0) {
                    a[i][j] = 'Q';
                    col[j] = v1[i + j] = v2[i - j + n] = 1;
                    self(self, i + 1);
                    col[j] = v1[i + j] = v2[i - j + n] = 0;
                    a[i][j] = '.';
                }
            }  
        };
        dfs(dfs, 0);
        return ans;
    }
};

[二叉树中的最大路径和]

题意：找到一棵树中的最大权值路径。

做法：树上 dp，类似于最大子段和的过程，维护每棵子树包含自身结点的最长向下路径，左右儿子最长向下路径 $max$ （注意分别与 $0$ 取 $max$，表示可以不选）加自身权值；然后每个结点更新答案，左右儿子最长向下路径和（注意分别与 $0$ 取 $max$，表示可以不选）加自身权值。

class Solution {
public:
    int maxPathSum(TreeNode* root) {
        int ans = -INT_MAX;
        auto dfs = [&](auto &self, TreeNode* t) -> int {
            if (!t) return 0;
            int lm = self(self, t->left);
            int rm = self(self, t->right);
            ans = max(ans, max(lm, 0) + max(rm, 0) + t->val);
            return max({lm, rm, 0}) + t->val;
        };
        dfs(dfs, root);
        return ans;
    }
};

class Solution {
public:
    string minWindow(string s, string t) {
        int mp[130] {};
        int fast = -1, slow = 0, less = 0;
        for (auto i: t) {
            if (mp[i] == 0) less++;
            mp[i]++;
        }

        int ansl = -1, ansr = s.size() - 1;
        while (fast < (int)s.size()) {
            while (less) {
                fast++;
                if (fast >= (int)s.size()) break;
                mp[s[fast]]--;
                if (mp[s[fast]] == 0) less--;
            }
            while (less == 0) {
                if (fast - slow < ansr - ansl) ansl = slow, ansr = fast;
                mp[s[slow]]++;
                if (mp[s[slow]] == 1) less++;
                slow++;
            }
        }
        if (ansl < 0) return "";
        else {
            string ans = "";
            for (int i = ansl; i <= ansr; i++) ans.push_back(s[i]);
            return ans;
        }
    }
};

class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        auto cmp = [&](ListNode* a, ListNode* b) -> int {
            return a->val > b->val;
        };
        priority_queue<ListNode*, vector<ListNode*>, decltype(cmp)> q(cmp);
        for (auto i: lists) {
            if (i) q.push(i);
        }
        ListNode* ans = new ListNode(0, nullptr);
        ListNode* ret = ans;
        while(!q.empty()) {
            auto node = q.top();
            q.pop();
            ans->next = node;
            ans = ans->next;
            node = node->next;
            if (node) q.push(node);
        }
        return ret->next;
    }
};

[最长有效括号]

class Solution {
public:
    int longestValidParentheses(string s) {
        int n = s.size(), ans = 0;
        vector<int> dp(n, 0);
        for (int i = 1; i < n; i++) {
            if (s[i] == '(') dp[i] = 0;
            else {
                if (s[i - 1] == '(') {
                    dp[i] = 2;
                    if (i - 2 >= 0) dp[i] = dp[i - 2] + 2;
                }
                else if (s[i - 1] == ')') {
                    if (i - dp[i - 1] - 1 >= 0 && s[i - dp[i - 1] - 1] =='(') {
                        dp[i] = dp[i - 1] + 2;
                        if (i - dp[i - 1] - 2 >= 0) dp[i] = dp[i - 1] + dp[i - dp[i - 1] - 2] + 2;
                    }
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
