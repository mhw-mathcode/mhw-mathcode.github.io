class Solution {
public:
    ListNode* middle(ListNode* head) {
        ListNode* lst = head;
        ListNode* slow = head;
        ListNode* fast = head;
        while (fast && fast->next) {
            lst = slow;
            slow = slow->next;
            fast = fast->next->next;
        }
        lst->next = nullptr;
        return slow;
    }
    ListNode* merge(ListNode* a, ListNode* b) {
        ListNode* ans = new ListNode(0, nullptr);
        ListNode* ans2 = ans;
        while (a && b) {
            if (a->val < b->val) {
                ans->next = a;
                a = a->next;
            }
            else {
                ans->next = b;
                b = b->next;
            }
            ans = ans->next;
        }
        ans->next = a ? a : b;
        return ans2->next;
    }
    ListNode* sortList(ListNode* head) {
        if (!head || !head->next) {
            return head;
        }
        ListNode* head2 = middle(head);
        head = sortList(head);
        head2 = sortList(head2);
        
        return merge(head, head2);
    }
};

class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int n = preorder.size();
        unordered_map<int, int> mp;
        // æ¯ä¸ªç‚¹åœ¨ä¸­åºéå†ä¸­çš„ä½ç½®
        for (int i = 0; i < n; i++) mp[inorder[i]] = i;

        auto dfs = [&](auto self, int pl, int pr, int il, int ir) ->  TreeNode* {
            if (pl > pr) return nullptr;
            int lsize = mp[preorder[pl]] - il;
            TreeNode *left = self(self, pl + 1, pl + lsize, il, il + lsize - 1);
            TreeNode *right = self(self, pl + lsize + 1, pr, il + lsize + 1, ir);
            return new TreeNode(preorder[pl], left, right);
        };

        return dfs(dfs, 0, n - 1, 0, n - 1);
    }   
};

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        if (nums1.size() > nums2.size()) swap(nums1, nums2);

        int sum = nums1.size() + nums2.size();
        nums1.insert(nums1.begin(), INT_MIN);
        nums2.insert(nums2.begin(), INT_MIN);
        nums1.push_back(INT_MAX);
        nums2.push_back(INT_MAX);        

        int l = 0, r = nums1.size() - 1, ans = r;
        while (l <= r) {
            int i = (l + r) / 2;
            int j = (sum + 1) / 2 - i; 
            if (nums1[i + 1] >= nums2[j]) r = i - 1, ans = min(ans, i);
            else l = i + 1;
        }
        int i = ans, j = (sum + 1) / 2 - i;
        
        if (sum & 1) return max(nums1[i], nums2[j]);
        else return (max(nums1[i], nums2[j]) + min(nums1[i + 1], nums2[j + 1])) / 2.0;
    }
};

class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            if (nums[i] < 1 || nums[i] > n) 
                nums[i] = n + 1;
        }
        for (int i = 0; i < n; i++) {
            if (abs(nums[i]) <= n) 
                nums[abs(nums[i]) - 1] = -abs(nums[abs(nums[i]) - 1]);
        }
        for (int i = 0; i < n; i++) {
            if (nums[i] >= 0) 
                return i + 1;
        }
        return n + 1;
    }
};

class MedianFinder {
public:
    priority_queue<int, vector<int>, greater<int>> qmax; // å°æ ¹å †ï¼Œè¾ƒå¤§çš„ä¸€åŠ
    priority_queue<int, vector<int>, less<int>> qmin; // å¤§æ ¹å †ï¼Œè¾ƒå°çš„ä¸€åŠ
    MedianFinder() { }
    
    void addNum(int num) {
        if (qmin.size() == 0) {
            qmin.push(num);
        }
        else if (num <= qmin.top())
        {
            qmin.push(num);
            if (qmin.size() - qmax.size() >= 2) {
                int t = qmin.top();
                qmin.pop();
                qmax.push(t);
            }
        }
        else {
            qmax.push(num);
            if (qmax.size() > qmin.size()) {
                int t = qmax.top();
                qmax.pop();
                qmin.push(t);
            }
        }
    }
    
    double findMedian() {
        if (qmin.size() == qmax.size()) return (qmin.top() + qmax.top()) / 2.0;
        else return qmin.top();
    }
};

class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        vector<int> col(n, 0), v1(2 * n, 0), v2(2 * n, 0);
        vector<string> a(n, string(n, '.'));
        vector<vector<string>> ans;
        auto dfs = [&](auto& self, int i) -> void {
            if (i == n) {
                ans.push_back(a);
                return;
            }
            for (int j = 0; j < n; j++) {
                if (col[j] == 0 && v1[i + j] == 0 && v2[i - j + n] == 0) {
                    a[i][j] = 'Q';
                    col[j] = v1[i + j] = v2[i - j + n] = 1;
                    self(self, i + 1);
                    col[j] = v1[i + j] = v2[i - j + n] = 0;
                    a[i][j] = '.';
                }
            }  
        };
        dfs(dfs, 0);
        return ans;
    }
};

class Solution {
public:
    int maxPathSum(TreeNode* root) {
        int ans = -INT_MAX;
        auto dfs = [&](auto &self, TreeNode* t) -> int {
            if (!t) return 0;
            int lm = self(self, t->left);
            int rm = self(self, t->right);
            ans = max(ans, max(lm, 0) + max(rm, 0) + t->val);
            return max({lm, rm, 0}) + t->val;
        };
        dfs(dfs, root);
        return ans;
    }
};

class Solution {
public:
    string minWindow(string s, string t) {
        int mp[130] {};
        int fast = -1, slow = 0, less = 0;
        for (auto i: t) {
            if (mp[i] == 0) less++;
            mp[i]++;
        }

        int ansl = -1, ansr = s.size() - 1;
        while (fast < (int)s.size()) {
            while (less) {
                fast++;
                if (fast >= (int)s.size()) break;
                mp[s[fast]]--;
                if (mp[s[fast]] == 0) less--;
            }
            while (less == 0) {
                if (fast - slow < ansr - ansl) ansl = slow, ansr = fast;
                mp[s[slow]]++;
                if (mp[s[slow]] == 1) less++;
                slow++;
            }
        }
        if (ansl < 0) return "";
        else {
            string ans = "";
            for (int i = ansl; i <= ansr; i++) ans.push_back(s[i]);
            return ans;
        }
    }
};

class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        auto cmp = [&](ListNode* a, ListNode* b) -> int {
            return a->val > b->val;
        };
        priority_queue<ListNode*, vector<ListNode*>, decltype(cmp)> q(cmp);
        for (auto i: lists) {
            if (i) q.push(i);
        }
        ListNode* ans = new ListNode(0, nullptr);
        ListNode* ret = ans;
        while(!q.empty()) {
            auto node = q.top();
            q.pop();
            ans->next = node;
            ans = ans->next;
            node = node->next;
            if (node) q.push(node);
        }
        return ret->next;
    }
};

class Solution {
public:
    int longestValidParentheses(string s) {
        int n = s.size(), ans = 0;
        vector<int> dp(n, 0);
        for (int i = 1; i < n; i++) {
            if (s[i] == '(') dp[i] = 0;
            else {
                if (s[i - 1] == '(') {
                    dp[i] = 2;
                    if (i - 2 >= 0) dp[i] = dp[i - 2] + 2;
                }
                else if (s[i - 1] == ')') {
                    if (i - dp[i - 1] - 1 >= 0 && s[i - dp[i - 1] - 1] =='(') {
                        dp[i] = dp[i - 1] + 2;
                        if (i - dp[i - 1] - 2 >= 0) dp[i] = dp[i - 1] + dp[i - dp[i - 1] - 2] + 2;
                    }
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
